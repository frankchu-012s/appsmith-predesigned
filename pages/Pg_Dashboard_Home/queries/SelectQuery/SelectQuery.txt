WITH directory AS (
  SELECT * FROM crm_views.customer_directory
), open_reviews AS (
  SELECT * FROM crm_views.manual_review_open
)
SELECT 'Active customers' AS metric, COUNT(*)::text AS value, '目前 active 客戶總數' AS note FROM directory WHERE status = 'active'
UNION ALL
SELECT 'Parent / root customers', COUNT(*)::text, '家庭/主要聯絡人數' FROM directory WHERE is_parent_customer = true
UNION ALL
SELECT 'Open reviews', COUNT(*)::text, '待人工審核項目' FROM open_reviews
UNION ALL
SELECT 'Recently assessed', COUNT(*)::text, '30 天內有健康評估' FROM directory WHERE latest_assessment_at >= now() - interval '30 days';
