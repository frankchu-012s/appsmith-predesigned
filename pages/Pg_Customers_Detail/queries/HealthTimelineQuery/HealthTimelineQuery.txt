WITH target AS (
  SELECT customer_id, customer_code FROM crm_views.customer_directory WHERE customer_code = '{{appsmith.URL.queryParams.customer_code || ""}}' LIMIT 1
)
SELECT 'assessment' AS event_type, assessment_date AS event_at, primary_concern AS title, severity_level AS status, symptoms AS details
FROM crm_core.health_assessments ha JOIN target t ON ha.customer_id = t.customer_id
UNION ALL
SELECT 'lifestyle', recorded_at, 'Lifestyle profile', sleep_quality, CONCAT_WS(' | ', exercise_habit, diet_patterns, stress_level)
FROM crm_core.lifestyle_profiles lp JOIN target t ON lp.customer_id = t.customer_id
UNION ALL
SELECT 'lab', test_date::timestamptz, 'Lab result', report_source, CONCAT_WS(' | ', 'HbA1c=' || hba1c, 'LDL=' || ldl, 'TG=' || triglycerides)
FROM crm_core.lab_results lr JOIN target t ON lr.customer_id = t.customer_id
ORDER BY event_at DESC NULLS LAST
LIMIT 50;
