SELECT
  customer_code,
  display_name,
  status,
  phone,
  email,
  family_role,
  is_parent_customer,
  source_channel,
  owner_user_label,
  last_contact_at,
  latest_submission_at,
  latest_assessment_at,
  open_review_count
FROM crm_views.customer_directory
WHERE customer_code = '{{appsmith.URL.queryParams.customer_code || ""}}'
LIMIT 1;
