SELECT
  customer_code,
  display_name,
  status,
  phone,
  email,
  family_role,
  is_parent_customer,
  source_channel,
  owner_user_label,
  last_contact_at,
  latest_submission_at,
  latest_assessment_at,
  open_review_count
FROM crm_views.customer_directory
WHERE (
  {{data_table.searchText ? "true" : "false"}} = false
  OR customer_code ILIKE '%{{data_table.searchText || ""}}%'
  OR display_name ILIKE '%{{data_table.searchText || ""}}%'
  OR COALESCE(phone, '') ILIKE '%{{data_table.searchText || ""}}%'
  OR COALESCE(email, '') ILIKE '%{{data_table.searchText || ""}}%'
)
ORDER BY {{['customer_code','display_name','status','latest_assessment_at','open_review_count'].includes(data_table.sortOrder.column) ? data_table.sortOrder.column : 'customer_code'}} {{data_table.sortOrder.order || 'ASC'}}
LIMIT {{data_table.pageSize || 25}}
OFFSET {{((data_table.pageNo || 1) - 1) * (data_table.pageSize || 25)}};
